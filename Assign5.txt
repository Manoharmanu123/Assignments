#remove elements from list(POP, REMOVE AND CLEAR)
print("{:-^40}".format("pop"))
l1 = list(range(1, 11))
print(f"l1:{l1}")
res = l1.pop(0)
print(f"res:{res}")
print(f"l1 :{l1}")
res1 = l1.pop(3)
print(f"res1:{res1}")
res2 = l1.pop()
print(f"res2:{res2}")
l1.pop()
print(f"l1:{l1}")
print("remove".center(40, "-"))
l2 = [1, 2, 3, 1, 2, 1, 3, 1, 1, 2, 3, 3, 2, 1]
print(f"l2 :{l2}")
res = l2.remove(1)
print(f"res:{res}")
l2.remove(3)
print(f"l2 :{l2}")
